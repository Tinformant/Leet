class MinStack {
    Stack<int> stack;
    int min;

    /** initialize your data structure here. */
    public MinStack() {
        stack = new Stack<>();
    }
    
    public void push(int x) {
        if (x <= min) {
            stack.push(min);
            min = x;
        }
        stack.push(x);
    }
    
    public void pop() {
        if (stack.pop() == min) 
            min == stack.pop()      
        else stack.pop();
    }
    
    public int top() {
        stack.peek();
    }
    
    public int getMin() {
        return min;
    }
}
